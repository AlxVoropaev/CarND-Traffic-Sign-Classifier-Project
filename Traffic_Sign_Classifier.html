import numpy as np
import scipy.ndimage as ndi
import cv2

class Dataset:
    
    def __init__(self, images, normalize = None, augment = True):
        self.images = images
        self.X = [self._get_features_(images[i]) for i in range(images.shape[0])]
        self.X = np.stack(self.X)
        if normalize:
            self.normalize = normalize
        else:
            self.normalize = (np.mean(self.X, axis=0), np.std(self.X, axis=0) + 0.001)
        self.X = (self.X - self.normalize[0]) / self.normalize[1]
        self.augment = augment
    
    def __len__(self):
        return self.X.shape[0]
    
    def __getitem__(self, item):
        return self._augment_(self.X[item])
        #return self.X[item]
    
    def get_image(self, idx):
        return self.images[idx]
    
    def _augment_(self, img):
        if not self.augment:
            return img
        if len(img.shape) == 4:
            # batch mode
            for i in range(img.shape[0]):
                img[i] = self._augment_(img[i])
            return img
        else:
            # rotation
            img = ndi.rotate(img, axes=(0, 1), angle=random.uniform(-5, 5), reshape=False, order=2)
            # shift 
            dx = random.uniform(-5, 5)
            dy = random.uniform(-5, 5)
            img = ndi.shift(img, shift=(dx, dy, 0))
            return img

    def _get_features_(self, img):
        img = self._fix_dark(img)
        
        # convert to float
        img = img.copy().astype('float32')
        img = img / 255.

        # normalize color
    
        # add Luv color space
        lab = cv2.cvtColor(img, cv2.COLOR_RGB2LAB)
        img = np.concatenate([img, lab], axis=2)

        return img
    
    def _fix_dark(self, img):
        lab = cv2.cvtColor(img, cv2.COLOR_RGB2LAB)
        lab_planes = cv2.split(lab)
        clahe = cv2.createCLAHE(clipLimit=2.0,tileGridSize=(4,4))    
        lab_planes[0] = clahe.apply(lab_planes[0])
        lab = cv2.merge(lab_planes)
        img = cv2.cvtColor(lab, cv2.COLOR_LAB2RGB)
        return img

import tensorflow as tf
from tensorflow.contrib.layers import flatten
import numpy


def _conv_layer(x, shape, name):
    sigma = 1 / numpy.sqrt((shape[0]*shape[1]*shape[2]))
    conv_w = tf.Variable(tf.random_uniform(shape, -sigma, sigma), name = name + '_w')
    conv_b = tf.Variable(tf.random_uniform([shape[-1]], -sigma, sigma), name = name + '_b')
    conv = tf.nn.conv2d(x, conv_w, strides=1, padding='SAME') + conv_b
    conv = tf.nn.relu(conv)
    conv = tf.nn.max_pool2d(conv, ksize=2, strides=2, padding='VALID')
    return conv
    
def lenet(x, n_classes, dropout):

    conv1 = _conv_layer(x, (3,3,6,32), 'conv1')
    conv2 = _conv_layer(conv1, (3,3,32,64), name='conv2')
    conv3 = _conv_layer(conv2, (3,3,64,128), name='conv3')
    conv4 = _conv_layer(conv3, (3,3,128,512), name='conv4')
    maxpool = tf.nn.max_pool2d(conv4, ksize=2, strides=2, padding='VALID')
    
    fc0 = flatten(maxpool)
    fc0 = tf.nn.dropout(fc0, rate=dropout)
    
    sigma = 1 / numpy.sqrt(512)
    fc1_w = tf.Variable(tf.random_uniform((512, 256), -sigma, sigma))
    fc1_b = tf.Variable(tf.random_uniform((256,), -sigma, sigma))
    fc1 = tf.matmul(fc0, fc1_w) + fc1_b
    fc1 = tf.nn.relu(fc1)
    fc1 = tf.nn.dropout(fc1, rate=dropout)
    
    sigma = 1 / numpy.sqrt(256)
    fc2_w = tf.Variable(tf.random_uniform((256, n_classes), -sigma, sigma))
    fc2_b = tf.Variable(tf.random_uniform((n_classes,), -sigma, sigma))
    fc2 = tf.matmul(fc1, fc2_w) + fc2_b
    
    return fc2

correct_answers = tf.equal(tf.argmax(logits, 1), tf.argmax(y_one_hot, 1))
accuracy_operation = tf.reduce_mean(tf.cast(correct_answers, tf.float32))

def evaluate(X_data, y_data):
    _EVAL_BATCH = 128
    total_samples = len(X_data)
    total_accuracy = 0.
    total_loss = 0.
    sess = tf.get_default_session()
    for offset in tqdm(range(0, total_samples, _EVAL_BATCH), total=total_samples/_EVAL_BATCH, desc='eval'):
        batch_x = X_data[offset:offset+_EVAL_BATCH]
        batch_y = y_data[offset:offset+_EVAL_BATCH]
        batch_accuracy, batch_loss = sess.run([accuracy_operation, loss_operation], 
                                              feed_dict={x: batch_x, y: batch_y, dropout:0})
        total_accuracy += batch_accuracy * len(batch_x)
        total_loss += batch_loss
    return {'acc': total_accuracy / total_samples, 'loss': total_loss}

Output Top 5 Softmax Probabilities For Each Image Found on the Web¶














For each of the new images, print out the model's softmax probabilities to show the certainty of the model's predictions (limit the output to the top 5 probabilities for each image). tf.nn.top_k could prove helpful here.


The example below demonstrates how tf.nn.top_k can be used to find the top k predictions for each image.


tf.nn.top_k will return the values and indices (class ids) of the top k predictions. So if k=3, for each sign, it'll return the 3 largest probabilities (out of a possible 43) and the correspoding class ids.


Take this numpy array as an example. The values in the array represent predictions. The array contains softmax probabilities for five candidate images with six possible classes. tf.nn.top_k is used to choose the three classes with the highest probability:
